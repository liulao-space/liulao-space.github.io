# this.props的应用 父传子

父组件中通过属性传值：props(Passing Data Through Props)
子组件中通过属性接收(this.props.[属性名])

代码示例：

import React, { Component } from 'react'

class Header extends Component{ // 子组件
    render(){
        return (
            <header style={{ width: '100%', height: 40, lineHeight: '40px', backgroundColor: this.props.bgc }}>
            {/* 子组件通过 this.props.title 接收对应标签属性值 */}
                { this.props.title}
                {/* 特殊用法： 专门获取子组件被调用，写成双标签时的子元素 */}
                { this.props.children}
            </header>
        )
    }
}

export default class App7 extends Component { // 父组件


    constructor(props){
        super(props)
        this.state = {
            title: '首页标题'
        }
    }

    render() {
        return (
        <div>
            {/* 父组件通过属性传值 */}
            <Header title={ this.state.title } bgc="#cfc" />
            <Header title="列表页页标题" bgc="pink" />
        </div>
        )
    }
}

# 特殊用法 this.props.children

父组件中调用子组件可以是双标签也可以是单标签，在使用父组件中使用双标签时，子组件可以通过 this.props.children 获取到被调用组件的子元素，如：

<Header title={ this.state.title } bgc="#cfc" />
<Header>这是双标签子元素</Header>
// this.props.children 获取到的值为： 这是双标签子元素

# props设置默认值

在父组件调用子组件中属性没有传递值时，可以在子组件中设置默认值，在 static defaultProps = {} 可添加默认属性值，如：

// 设置属性默认值 
    static defaultProps = {
        title: '默认标题',
        bgc: 'red'

# state与props小结 (注意点、区别)

1、定义组件的时候，状态初始化在 constructor 方法中完成

constructor(props){
    // 创建组件会执行的构造函数
    super(props) // 如果类继承其他类 需要加上 super()
    // 定义一个组件的状态数据
    this.state = {
        // 状态属性名：值
        num: 20
    }
}

# 拓展 key 值的作用

key的值一般设置为数据的id值(唯一)，有利于提高更新的效率，减少不必要的DOM操作

import React, { Component } from 'react'
// key值拓展 提高更新的速率
export default class App8 extends Component {

    constructor(props){
        super(props)
        this.state = {
            list:[
                {id: 1, value: '值1'},
                {id: 2, value: '值2'},
                {id: 3, value: '值3'},
            ]
        }
    }

    handleClick(){
        this.setState({
            list: [
                {id: 4, value: '值4'},
                {id: 1, value: '值1'},
                {id: 2, value: '值2'},
                {id: 3, value: '值3'},
            ]
        })
    }

    render() {
        return (
        <div>
            <button onClick={this.handleClick.bind(this)}>添加一项</button>
            <ul>
                {
                    this.state.list.map((v, k) => {
                        return(
                            // key的值一般设置为数据id的值 有利于提高更新的效率 减少不必要的DOM操作
                            // key=k 整个列表都会重新渲染 而不是仅仅插入一条数据
                            // key = v.id 数据没有变化的视图就不会重新渲染
                            <li key={v.id}>{v.value}</li>
                        )
                    })
                }
            </ul>
        </div>
        )
    }
}

import React, { Component } from 'react'
import PropTypes from 'prop-types'

class Header extends Component{
    // 属性值类型提示 会有一个警告

# 使用 Context 进行props属性值的多级传递

React 组件之间的通信是基于 props 的数据传递，数据需要一级一级从上到下传递。如果组件级别过多传递就会过于麻烦。React中context配置可以解决组件跨级值传递。

1、在父组件中声明数据类型和值

import React, {Component} from 'react'
import Proptypes from 'prop-types' // 安装的插件

class Child extends Component{ // 子组件
    render(){
        return (
            <ChildChild />
        )
    }
}

export default App extends Component{ // 父组件
    // 1、在父组件中声明数据类型和值
    static childContextTypes = {
        // 属性类型
        title: Proptypes.string
    }
    // 2、将要传递给子孙组件的属性，写到 getChildContext 方法返回的对象中
    getChildContext(){
        return {
            title: '标题'
        }
    }
    render(){
        retrun(
            <Child />
        )
    }
}

2、在孙（多级）组件中声明需要的数据类型

在无论“那一级”子组件中，只需声明需要的全局上下文数据，就可自动注入到子组件的 context 属性中

class ChildChild extends Component{

    static contextTypes = {
        title: PropTypes.string
    }

    render(){
        return(
            // 通过 this.context.属性名 获取数据
            <div>{this.context.title}</div>
        )
    }
}

# 子组件向父组件传值

在父组件中传递一个属性方法，接收参数；在子组件中通过 this.props.方法名 调用父组件中传递过来的方法，传入需要的参数值即可实现子传父。

import React, { Component } from 'react'

// 子组件向 父组件传值
class Child extends Component{

    constructor(props) {
        super(props)
        this.state = {
            // 给子组件赋初始值 父组件传过来的属性值
            count: this.props.children
        }
    }

    handleClick(){
        this.setState({
            count: this.state.count+1
        })
        // 接收父组件传递过来的方法
        this.props.fatherChange(this.state.count)
    }

    render(){
        return(
            <div>
                {/* <p>{this.props.children}</p>  */}
                <p>{this.state.count}</p> 
                <button onClick={this.handleClick.bind(this)}>点击</button>
            </div>
            
        )
    }
}

export default class App11 extends Component {
    constructor(props){
        super(props)
        this.state = {
            num: 10
        }
    }

    // 向子组件传递一个方法实现子组件向父组件传值
    fatherChange(value){
        this.setState({
            num: value
        })
    }

    render() {
        return (
        <Child fatherChange={this.fatherChange.bind(this)}>{this.state.num}</Child>
        )
    }
}
